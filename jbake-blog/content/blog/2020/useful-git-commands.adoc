.rebase last ${NUM} commits. Wit that you can squash them to one commit 
[source, bash]
----
git rebase -i HEAD~${NUM}
----

.compare file with last commit
[source, sh]
----
git diff  HEAD^ HEAD  -- {FILE}
----

.revert a single file
[source, sh]
----
git checkout -- {FILE}
----

.show git graph
[source, shell]
----
git log --graph --decorate --oneline --all
----
